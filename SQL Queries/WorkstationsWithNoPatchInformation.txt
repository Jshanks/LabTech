SELECT DISTINCT clients.name as Client,locations.name AS Location,computers.name as Computer,computers.computerid as ComputerID,computers.lastcontact as LastContact FROM computers,clients,locations WHERE computers.clientid=clients.clientid AND computers.locationid=locations.locationid AND computers.computerid NOT IN (SELECT DISTINCT computerid FROM v_hotfixes) AND OS LIKE '%Windows%' AND OS NOT LIKE '%server%'
